javascript设计模式

发表于 2015-06-21   |  
  • 基本constructor（构造器）

代码：

function Car(model,year,miles) {
    this.model = model;
    this.year = year;
    this.miles = miles;
    this.toString = function() {
        return this.model + ' has done ' + this.miles + ' miles';
    }
}
var car= new Car('红旗',10,20000);
console.log(car.toString());

缺点是，这个构造器实例化的对象的函数方法不能公用，每个新对象都有一个自己的方法

  • 带原型的constructor(构造器)

把上面的代码改为

function Car(model,year,miles) {
    this.model = model;
    this.year = year;
    this.miles = miles;
}
Car.prototype.toString = function() {
    return this.model + ' has done ' + this.miles + ' miles';
}
var car1 = new Car('benz',5,30000);
var car2 = new Car('bmw',6,50000);
console.log(car1.toString());
console.log(car2.toString());

这次的代码，car1,car2是用的一个toString方法，实现了所有实例共享一个方法

  • 使用对象字面量表示法定义模块

代码：

var myModule = {
    //基本属性
    myProperty: 'someValue',
    //对象字面量可以包含属性和方法
    //例如：可以声明模块的配置对象
    myConfig: {
        useCaching: true,
        language:'zh'
    },
    //基本方法
    myMothed: function() {
        console.log('hello world!');
    },
    //根据当前配置输出信息
    myMothed2: function() {
        console.log('Caching is:' + (this.myConfig.useCaching) ? 'enabled' : 'dissabled');
    },
    //重写当前的配置
    myMothed3: function(newConfig) {
        if(typeof newConfig === 'object') {
            this.myConfig = newConfig;
            console.log(this.myConfig.language);
        }
    }
};
//使用
console.log(myModule.myMothed());//hello wrold
console.log(myModule.myMothed2());//enabled;
console.log(myModule.myMothed3({language:'en',useCaching: false})); //en

使用对象字面量有利于封装和组织代码

  • Module(模块)模式
    能够使一个单独的对象拥有公有/私有的属性和方法，从而屏蔽来自全局作用域的特殊部分，产生的结果是：函数名与在页面上其它脚本定义的函数冲突的可能性降低；Module模式使用闭包封装‘私有’状态和组织，防止其泄露至全局作用域，并与别的开发者的接口发生冲突。通过该模式，只需返回一个公有API，其他的一切则都维持在私有闭包里。该模式除了返回一个对象而不是一个函数外，非常类似一个立即调用的函数表达式。

一个自包含的Module：

var myModule = (function() {
    var counter = 0;
    return {
        incrementCounter:function() {
            return ++counter;
        },
        resetCounter:function() {
            console.log('counter value prior to reset:' + counter);
            counter = 0;
        }
    }
    })();
    console.log(myModule.incrementCounter());

counter变量只能通过返回的两个函数来操作;
一个包含命名空间、公有、私有变量的Module：

var myNamespace = (function() {
    //私有计数器变量
    var myPrivateVar = 0;
    //记录所有参数的私有函数
    var myPrivateMethod = function(foo) {
        console.log(foo);
    };
    return {
        //公有变量
        myPublicVar: "foo",
        //调用私有变量和方法的公有函数
        myPublicFunction: function(bar) {
            //增加私有计数器的值
            myPrivateVar++;
            //传入bar调用私有方法
            myPrivateMethod(bar);
        }
    };
    })();
    console.log(myNamespace.myPublicVar);//foo
    console.log(myNamespace.myPublicFunction('hello world!'));//hello world!
  • Module模式变化
    • 引入混入
      模式的这种变化演示了全局变量（如：jQuery）如何作为参数传递给模块的匿名函数。并允许我们按照我们所希望的为他们取个本地别名；

代码：

var moModule = (function(jQ,_) {
    function privateMethod () {
        jQ(".container").html('test');
    }
    function privateMethod2() {
        console.log(_.min([10.5,100,2,1000]));
    }
    return {
        publicMethod:function() {
            privateMethod();
        }
    };
    })(jQuery,_);//引入jQuery和underscore
    myModule.publicMethod();
  • 引出
    允许我们声明全局变量而不需要实现他们，并且同样地支持上一个实例中的全局引入的概念；

代码：

var myModule = (function() {
    //模块对象
    var module = {},
        privateVar = 'hello world1';
        function privateMethod() {
            console.log('foo');
        }
        module.publicProperty = 'foobar';
        module.publicMethod = function() {
            console.log(privateVar);
        };
        return module;
    })();
    console.log(myModule.publicProperty);//foobar
  • Revealing Module(揭示模块)模式
    Module模式的改进版，能够在私有范围内简单定义所有的函数和变量，并返回一个匿名对象，它拥有指向私有函数的指针

代码：

var myRevealingModule = function() {
    var privateVar = 'Ben Cherry',
        publicVar = "Hey there";
    function privateFunction() {
        console.log('Name:'+ privateVar);
    }
    function privateSetName(strName) {
        privateVar = strName;
    }
    function publicGetName() {
        privateFunction();
    }
    //将暴露的公有指针指向到私有函数和属性上
    return {
        setName: publicSetName,
        greeting: publicVar,
        getName: publicGetName
    };
}
  • Singleton(单例)模式
    类的实例化次数只能是一次

代码：

var mySingleton = (function() {
    //实例保持了Singleton的一个引用
    var instance;
    function init() {
        //singleton
        //私有方法和变量
        function privateMethod() {
            console.log('i am private');
        }
        var privateVar = 'i am also private';
        var privateRandomNum = Math.random();
        return {
            publicMethod: function() {
                console.log('the public can see me!');
            },
            publicProperty: 'i am also public',
            getRandomNum:function() {
                return privateRandomNum;
            }
        };
    }
    return {
        //获取singleton的实例，如果存在就返回，不存在就创建新实例
        getInstance: function() {
            if(!instance) {
                instance = init();
            }
            return instance;
        }
    };
    })();
    var singleA = mySingleton.getInstance();
    var singleB = mySingleton.getInstance();
    console.log(singleA === singleB);
